Supabase SQL Setup
Run this SQL in your Supabase project → SQL Editor → New Query: 
-- 1. Changelog entries table
CREATE TABLE IF NOT EXISTS changelog_entries (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  version     TEXT NOT NULL,
  title       TEXT NOT NULL,
  date        DATE NOT NULL,
  description TEXT,
  code        TEXT,
  tags        TEXT[]  DEFAULT '{}',
  bullets     TEXT[]  DEFAULT '{}',
  features    TEXT[]  DEFAULT '{}',
  bugs        TEXT[]  DEFAULT '{}',
  media       JSONB   DEFAULT '[]',
  status      TEXT    NOT NULL DEFAULT 'draft',
  created_at  TIMESTAMPTZ DEFAULT now(),
  updated_at  TIMESTAMPTZ DEFAULT now()
);

-- 2. Auto-update updated_at
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN NEW.updated_at = now(); RETURN NEW; END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER changelog_updated_at
  BEFORE UPDATE ON changelog_entries
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

-- 3. Row Level Security — only authenticated users can write
ALTER TABLE changelog_entries ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Public read published"
  ON changelog_entries FOR SELECT
  USING (status = 'published' OR auth.role() = 'authenticated');

CREATE POLICY "Auth users can insert"
  ON changelog_entries FOR INSERT
  WITH CHECK (auth.role() = 'authenticated');

CREATE POLICY "Auth users can update"
  ON changelog_entries FOR UPDATE
  USING (auth.role() = 'authenticated');

CREATE POLICY "Auth users can delete"
  ON changelog_entries FOR DELETE
  USING (auth.role() = 'authenticated');

-- 4. Storage bucket for media
INSERT INTO storage.buckets (id, name, public)
VALUES ('changelog-media', 'changelog-media', true)
ON CONFLICT DO NOTHING;

CREATE POLICY "Public read media"
  ON storage.objects FOR SELECT
  USING (bucket_id = 'changelog-media');

CREATE POLICY "Auth upload media"
  ON storage.objects FOR INSERT
  WITH CHECK (bucket_id = 'changelog-media' AND auth.role() = 'authenticated');

CREATE POLICY "Auth delete media"
  ON storage.objects FOR DELETE
  USING (bucket_id = 'changelog-media' AND auth.role() = 'authenticated');
